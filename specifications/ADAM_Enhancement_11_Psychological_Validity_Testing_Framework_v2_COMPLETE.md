                validation_status=ValidationStatus.INSUFFICIENT_DATA

        matched = data.matched_exposures
        control = data.control_exposures
        
        if len(matched) < self.config.minimum_matched_samples:
            logger.warning(f"Insufficient matched samples for {mechanism}")

            logger.warning(f"Insufficient control samples for {mechanism}")

        control_conversions = sum(1 for e in control if e.outcome_value > 0)

        control_rate = control_conversions / len(control) if control else 0

        absolute_lift = matched_rate - control_rate
        relative_lift = (matched_rate - control_rate) / control_rate if control_rate > 0 else 0
        
        # Statistical test (chi-square)
        contingency = [
            [matched_conversions, len(matched) - matched_conversions],
            [control_conversions, len(control) - control_conversions]
        ]

            chi2, p_value, dof, expected = stats.chi2_contingency(contingency)

            chi2, p_value = 0.0, 1.0

            matched_rate * (1 - matched_rate) / len(matched) +
            control_rate * (1 - control_rate) / len(control)
        ) if matched and control else 0

        lift_ci = (absolute_lift - z * se, absolute_lift + z * se)

            test_statistic=chi2,
            p_value=p_value,
            degrees_of_freedom=1,
            effect_size=effect_size,
            effect_size_metric=EffectSizeMetric.COHENS_H,
            confidence_interval=lift_ci,
            sample_size=len(matched) + len(control),
            is_significant=p_value < self.config.significance_level,
            alpha=self.config.significance_level
        )

            status = ValidationStatus.PASSED

            status = ValidationStatus.WARNING

    def validate_internal_consistency(
        self,
        construct_name: str,
        item_scores: List[List[float]]
    ) -> Tuple[float, ValidationStatus]:
        """
        Calculate Cronbach's alpha for internal consistency.
        
        Args:
            construct_name: Name of the construct
            item_scores: List of item scores for each respondent
                        Shape: (n_respondents, n_items)
        
        Returns:
            Tuple of (alpha, status)
        """
        if len(item_scores) < 10:
            return 0.0, ValidationStatus.INSUFFICIENT_DATA
        
        item_scores = np.array(item_scores)
        n_items = item_scores.shape[1]
        
        # Item variances
        item_variances = np.var(item_scores, axis=0, ddof=1)
        
        # Total score variance
        total_scores = np.sum(item_scores, axis=1)
        total_variance = np.var(total_scores, ddof=1)
        
        # Cronbach's alpha
        if total_variance == 0:
            return 0.0, ValidationStatus.FAILED
        
        alpha = (n_items / (n_items - 1)) * (1 - np.sum(item_variances) / total_variance)
        
        # Determine status
        if alpha >= self.config.minimum_alpha:
            status = ValidationStatus.PASSED
        elif alpha >= 0.60:
            status = ValidationStatus.WARNING
        else:
            status = ValidationStatus.FAILED
        
        return alpha, status

        late_values = [v for _, v in all_measurements[midpoint:]]
        
        # Compare means
        early_mean = np.mean(early_values)
        late_mean = np.mean(late_values)
        drift_magnitude = late_mean - early_mean
        
        # Statistical test
        t_stat, p_value = stats.ttest_ind(early_values, late_values)
        
        # Effect size
        pooled_std = np.sqrt((np.var(early_values) + np.var(late_values)) / 2)
        cohens_d = abs(drift_magnitude) / pooled_std if pooled_std > 0 else 0
        
        drift_detected = p_value < 0.01 and cohens_d > 0.2
        
        return {
            "drift_detected": drift_detected,
            "early_mean": early_mean,
            "late_mean": late_mean,
            "drift_magnitude": drift_magnitude,
            "drift_direction": "increasing" if drift_magnitude > 0 else "decreasing",
            "t_statistic": t_stat,
            "p_value": p_value,
            "cohens_d": cohens_d,
            "n_early": len(early_values),
            "n_late": len(late_values),
            "insufficient_data": False
        }
    
    async def validate_construct_temporal(
        self,
        construct: str
    ) -> TemporalValidationResult:
        """Validate temporal patterns for a single construct."""
        expected = self.EXPECTED_STABILITY.get(construct, {})
        construct_type = expected.get("type", "trait")
        
        # Test-retest at multiple intervals
        retest_intervals = [f"{d}_days" for d in self.config.retest_intervals_days]
        retest_correlations = {}
        
        for interval_days in self.config.retest_intervals_days:
            result = self.calculate_test_retest(construct, interval_days)
            if result.get("correlation") is not None:
                retest_correlations[f"{interval_days}_days"] = result["correlation"]
        
        # Calculate ICC for traits
        trait_icc = None
        trait_sem = None
        
        if construct_type == "trait":
            icc_result = self.calculate_icc(construct)
            trait_icc = icc_result.get("icc")
            
            if trait_icc is not None and icc_result.get("within_variance") is not None:
                trait_sem = np.sqrt(icc_result["within_variance"])
        
        # Calculate variability for states
        state_variability = None
        state_responsiveness = None
        
        if construct_type == "state":
            var_result = self.calculate_within_person_variability(construct)
            state_variability = var_result.get("mean_variability")
        
        # Drift detection
        drift_result = self.detect_drift(construct, self.config.drift_window_days)
        drift_detected = drift_result.get("drift_detected", False)
        drift_magnitude = drift_result.get("drift_magnitude") if drift_detected else None
        drift_direction = drift_result.get("drift_direction") if drift_detected else None
        
        # Determine status
        if construct_type == "trait":
            min_icc = expected.get("min_icc", 0.70)
            if trait_icc is not None and trait_icc >= min_icc and not drift_detected:
                status = ValidationStatus.PASSED
            elif trait_icc is not None and trait_icc >= 0.60:
                status = ValidationStatus.WARNING
            else:
                status = ValidationStatus.FAILED
        else:  # state
            min_var = expected.get("min_variability", 0.10)
            max_var = expected.get("max_variability", 0.50)
            
            if state_variability is not None and min_var <= state_variability <= max_var:
                status = ValidationStatus.PASSED
            elif state_variability is not None:
                status = ValidationStatus.WARNING
            else:
                status = ValidationStatus.FAILED
        
        return TemporalValidationResult(
            construct_name=construct,
            construct_type=construct_type,
            retest_intervals=retest_intervals,
            retest_correlations=retest_correlations,

            trait_sem=trait_sem,
            state_variability=state_variability,
            state_responsiveness=state_responsiveness,
            drift_detected=drift_detected,
            drift_magnitude=drift_magnitude,
            drift_direction=drift_direction,
            validation_status=status
        )
    
    async def validate_all(self) -> Dict[str, TemporalValidationResult]:

"""
ADAM Enhancement #11: Psychological Validity Testing Framework
Fairness Validation Engine.

Key question: Are ADAM's psychological insights fair across demographic groups?
"""

from datetime import datetime
from typing import Dict, List, Optional, Tuple, Any
import numpy as np
from scipy import stats
from collections import defaultdict
import logging

logger = logging.getLogger(__name__)


class FairnessValidator:
    """
    Validates fairness of ADAM's psychological predictions and recommendations.
    
    Key validations:
    1. Accuracy parity across demographic groups
    2. Mechanism effectiveness parity
    3. Measurement invariance (do constructs mean the same thing across groups?)
    4. Outcome equity
    """
    
    def __init__(
        self,
        config: FairnessValidationConfig,
        graph_client: 'Neo4jClient'
    ):
        self.config = config
        self.graph = graph_client
        
        # Data storage
        self.demographic_data: Dict[str, Dict[str, str]] = {}  # user_id -> demographics
        self.predictions: Dict[str, Dict[str, float]] = {}  # user_id -> predictions
        self.outcomes: Dict[str, List[Dict]] = defaultdict(list)  # user_id -> outcomes
    
    async def load_validation_data(
        self,
        start_date: datetime,
        end_date: datetime
    ) -> int:
        """Load demographic and outcome data for fairness validation."""
        query = """
        MATCH (u:User)
        WHERE u.demographics IS NOT NULL
        OPTIONAL MATCH (u)-[:HAS_PREDICTION]->(p:PersonalityPrediction)
        WHERE p.timestamp >= $start_date AND p.timestamp <= $end_date
        OPTIONAL MATCH (u)-[exp:EXPOSED_TO]->(cv:CopyVariant)-[:RESULTED_IN]->(out:Outcome)
        WHERE exp.timestamp >= $start_date AND exp.timestamp <= $end_date
        RETURN 
            u.user_id as user_id,
            u.demographics as demographics,
            p as prediction,
            collect({
                mechanism: cv.mechanism,
                outcome_type: out.outcome_type,
                outcome_value: out.value
            }) as outcomes
        """
        
        results = await self.graph.execute_query(
            query,
            {"start_date": start_date.isoformat(), "end_date": end_date.isoformat()}
        )
        
        for record in results:
            user_id = record["user_id"]
            
            if record["demographics"]:
                self.demographic_data[user_id] = record["demographics"]
            
            if record["prediction"]:
                pred = record["prediction"]
                self.predictions[user_id] = {
                    "openness": pred.get("openness"),
                    "conscientiousness": pred.get("conscientiousness"),
                    "extraversion": pred.get("extraversion"),
                    "agreeableness": pred.get("agreeableness"),
                    "neuroticism": pred.get("neuroticism")
                }
            
            for outcome in record["outcomes"]:
                if outcome.get("outcome_value") is not None:
                    self.outcomes[user_id].append(outcome)
        
        logger.info(f"Loaded fairness data for {len(self.demographic_data)} users")
        return len(self.demographic_data)
    
    def calculate_group_accuracy(
        self,
        attribute: DemographicAttribute,
        ground_truth: Dict[str, Dict[str, float]]
    ) -> Dict[str, Dict[str, float]]:
        """
        Calculate prediction accuracy for each demographic group.
        
        Args:
            attribute: Demographic attribute to segment by
            ground_truth: Dict mapping user_id to ground truth personality
        """
        attr_name = attribute.value
        group_accuracies = defaultdict(lambda: {"predictions": [], "actuals": []})
        
        for user_id, truth in ground_truth.items():
            if user_id not in self.demographic_data:
                continue
            if user_id not in self.predictions:
                continue
            
            group = self.demographic_data[user_id].get(attr_name)
            if not group:
                continue
            
            pred = self.predictions[user_id]
            
            # Calculate mean absolute error across traits
            errors = []
            for trait in ["openness", "conscientiousness", "extraversion", 
                         "agreeableness", "neuroticism"]:
                if pred.get(trait) is not None and truth.get(trait) is not None:
                    errors.append(abs(pred[trait] - truth[trait]))
            
            if errors:
                group_accuracies[group]["predictions"].append(np.mean(errors))
        
        # Calculate final accuracies per group
        result = {}
        for group, data in group_accuracies.items():
            if len(data["predictions"]) >= 30:
                mean_error = np.mean(data["predictions"])
                result[group] = {
                    "mean_absolute_error": mean_error,
                    "accuracy": 1 - mean_error / 4,  # Normalize to 0-1 (4 is max error on 1-5 scale)
                    "n_samples": len(data["predictions"])
                }
        
        return result
    
    def calculate_mechanism_effectiveness_by_group(
        self,
        attribute: DemographicAttribute
    ) -> Dict[str, Dict[str, float]]:
        """
        Calculate mechanism effectiveness for each demographic group.
        """
        attr_name = attribute.value
        group_mechanism_outcomes = defaultdict(lambda: defaultdict(list))
        
        for user_id, outcome_list in self.outcomes.items():
            if user_id not in self.demographic_data:
                continue
            
            group = self.demographic_data[user_id].get(attr_name)
            if not group:
                continue
            
            for outcome in outcome_list:
                mechanism = outcome.get("mechanism")
                value = outcome.get("outcome_value", 0)
                
                if mechanism:
                    group_mechanism_outcomes[group][mechanism].append(value)
        
        # Calculate conversion rates per group per mechanism
        result = {}
        for group, mechanisms in group_mechanism_outcomes.items():
            result[group] = {}
            for mechanism, values in mechanisms.items():
                if len(values) >= 30:
                    result[group][mechanism] = np.mean([1 if v > 0 else 0 for v in values])
        
        return result
    
    def calculate_disparate_impact(
        self,
        attribute: DemographicAttribute
    ) -> Dict[str, float]:
        """
        Calculate disparate impact ratio for outcomes.
        
        Disparate impact = (rate for disadvantaged group) / (rate for advantaged group)
        Must be >= 0.80 to pass the "80% rule"
        """
        attr_name = attribute.value
        group_outcomes = defaultdict(list)
        
        for user_id, outcome_list in self.outcomes.items():
            if user_id not in self.demographic_data:
                continue
            
            group = self.demographic_data[user_id].get(attr_name)
            if not group:
                continue
            
            # Aggregate positive outcomes
            positive_outcomes = sum(1 for o in outcome_list if o.get("outcome_value", 0) > 0)
            total_outcomes = len(outcome_list)
            
            if total_outcomes > 0:
                group_outcomes[group].append(positive_outcomes / total_outcomes)
        
        # Calculate rates per group
        group_rates = {}
        for group, rates in group_outcomes.items():
            if len(rates) >= 30:
                group_rates[group] = np.mean(rates)
        
        if len(group_rates) < 2:
            return {"ratio": None, "insufficient_groups": True}
        
        # Disparate impact = min_rate / max_rate
        min_rate = min(group_rates.values())
        max_rate = max(group_rates.values())
        
        ratio = min_rate / max_rate if max_rate > 0 else 0
        
        return {
            "ratio": ratio,
            "passes_80_percent_rule": ratio >= 0.80,
            "group_rates": group_rates,
            "min_group": min(group_rates, key=group_rates.get),
            "max_group": max(group_rates, key=group_rates.get)
        }
    
    def test_measurement_invariance(
        self,
        attribute: DemographicAttribute,
        ground_truth: Dict[str, Dict[str, float]]
    ) -> Dict[str, bool]:
        """
        Test measurement invariance across demographic groups.
        
        Simplified version: Check if trait intercorrelations are similar across groups.
        Full version would use multi-group CFA.
        """
        attr_name = attribute.value
        group_data = defaultdict(lambda: {trait: [] for trait in 
            ["openness", "conscientiousness", "extraversion", "agreeableness", "neuroticism"]})
        
        for user_id, truth in ground_truth.items():
            if user_id not in self.demographic_data:
                continue
            
            group = self.demographic_data[user_id].get(attr_name)
            if not group:
                continue
            
            for trait in ["openness", "conscientiousness", "extraversion", 
                         "agreeableness", "neuroticism"]:
                if truth.get(trait) is not None:
                    group_data[group][trait].append(truth[trait])
        
        # Calculate correlation matrices for each group
        group_correlations = {}
        for group, trait_data in group_data.items():
            if all(len(trait_data[t]) >= 50 for t in trait_data):
                corr_matrix = np.corrcoef([trait_data[t] for t in 
                    ["openness", "conscientiousness", "extraversion", "agreeableness", "neuroticism"]])
                group_correlations[group] = corr_matrix
        
        if len(group_correlations) < 2:
            return {"configural": None, "metric": None, "scalar": None}
        
        # Compare correlation matrices (simplified invariance test)
        matrices = list(group_correlations.values())
        max_diff = 0
        for i in range(len(matrices)):
            for j in range(i + 1, len(matrices)):
                diff = np.max(np.abs(matrices[i] - matrices[j]))
                max_diff = max(max_diff, diff)
        
        # Thresholds for invariance
        configural = max_diff < 0.30  # Same factor structure
        metric = max_diff < 0.15  # Similar factor loadings
        scalar = max_diff < 0.10  # Similar intercepts
        
        return {
            "configural": configural,
            "metric": metric,
            "scalar": scalar,
            "max_correlation_difference": max_diff
        }
    
    async def validate_attribute(
        self,
        attribute: DemographicAttribute,
        ground_truth: Optional[Dict[str, Dict[str, float]]] = None
    ) -> FairnessValidationResult:
        """Validate fairness for a single demographic attribute."""
        
        # Get groups
        groups = list(set(
            self.demographic_data[uid].get(attribute.value)
            for uid in self.demographic_data
            if self.demographic_data[uid].get(attribute.value)
        ))
        
        # Group accuracies (if ground truth available)
        if ground_truth:
            group_accuracies = self.calculate_group_accuracy(attribute, ground_truth)
        else:
            group_accuracies = {}
        
        accuracy_values = [g.get("accuracy", 0) for g in group_accuracies.values()]
        accuracy_disparity = max(accuracy_values) - min(accuracy_values) if accuracy_values else 0
        
        # Mechanism effectiveness by group
        group_mechanism_effectiveness = self.calculate_mechanism_effectiveness_by_group(attribute)
        
        # Calculate mechanism disparity
        mechanism_disparity = {}
        all_mechanisms = set()
        for group_mechs in group_mechanism_effectiveness.values():
            all_mechanisms.update(group_mechs.keys())
        
        for mechanism in all_mechanisms:
            rates = [
                group_mechs.get(mechanism, 0)
                for group_mechs in group_mechanism_effectiveness.values()
                if mechanism in group_mechs
            ]
            if len(rates) >= 2:
                mechanism_disparity[mechanism] = max(rates) - min(rates)
        
        # Measurement invariance
        if ground_truth:
            invariance = self.test_measurement_invariance(attribute, ground_truth)
        else:
            invariance = {"configural": None, "metric": None, "scalar": None}
        
        # Disparate impact
        di_result = self.calculate_disparate_impact(attribute)
        disparate_impact_ratio = di_result.get("ratio", 0) or 0
        passes_80_rule = di_result.get("passes_80_percent_rule", False)
        
        # Equal opportunity (simplified: similar accuracy across groups)
        equal_opportunity_ratio = 1 - accuracy_disparity if accuracy_disparity < 1 else 0
        
        # Demographic parity ratio
        group_rates = di_result.get("group_rates", {})
        if group_rates:
            demographic_parity_ratio = min(group_rates.values()) / max(group_rates.values()) if max(group_rates.values()) > 0 else 0
        else:
            demographic_parity_ratio = 0
        
        # Calculate fairness score
        fairness_components = []
        if disparate_impact_ratio > 0:
            fairness_components.append(min(disparate_impact_ratio / 0.80, 1.0))
        if accuracy_disparity < 1:
            fairness_components.append(1 - accuracy_disparity / 0.20)  # Target < 0.20
        
        fairness_score = np.mean(fairness_components) if fairness_components else 0.5
        
        # Flag issues
        flagged_issues = []
        if not passes_80_rule:
            flagged_issues.append(f"Disparate impact ratio {disparate_impact_ratio:.2f} < 0.80")
        if accuracy_disparity > 0.10:
            flagged_issues.append(f"Accuracy disparity {accuracy_disparity:.2f} > 0.10")
        for mech, disp in mechanism_disparity.items():
            if disp > 0.15:
                flagged_issues.append(f"Mechanism {mech} disparity {disp:.2f} > 0.15")
        
        # Determine status
        if len(flagged_issues) == 0:
            status = ValidationStatus.PASSED
        elif len(flagged_issues) <= 2:
            status = ValidationStatus.WARNING
        else:
            status = ValidationStatus.FAILED
        
        return FairnessValidationResult(
            attribute=attribute,
            groups=groups,
            group_accuracies=group_accuracies,
            accuracy_disparity=accuracy_disparity,
            group_mechanism_effectiveness=group_mechanism_effectiveness,
            mechanism_disparity=mechanism_disparity,
            configural_invariance=invariance.get("configural"),
            metric_invariance=invariance.get("metric"),
            scalar_invariance=invariance.get("scalar"),
            demographic_parity_ratio=demographic_parity_ratio,
            equal_opportunity_ratio=equal_opportunity_ratio,
            disparate_impact_ratio=disparate_impact_ratio,
            passes_80_percent_rule=passes_80_rule,
            fairness_score=fairness_score,
            validation_status=status,
            flagged_issues=flagged_issues
        )
    
    async def validate_all(
        self,
        ground_truth: Optional[Dict[str, Dict[str, float]]] = None
    ) -> Dict[DemographicAttribute, FairnessValidationResult]:
        """Validate fairness across all configured attributes."""
        results = {}
        
        for attribute in self.config.protected_attributes:
            logger.info(f"Validating fairness for: {attribute.value}")
            results[attribute] = await self.validate_attribute(attribute, ground_truth)
        
        return results
    
    def generate_report(

"""
ADAM Enhancement #11: Psychological Validity Testing Framework
Calibration Validation Engine.

Key question: When ADAM says "80% confident", is it correct 80% of the time?
"""

from datetime import datetime
from typing import Dict, List, Optional, Tuple, Any
import numpy as np
from scipy import stats
from dataclasses import dataclass, field
import logging

logger = logging.getLogger(__name__)


@dataclass
class CalibrationBin:
    """Data for a single calibration bin."""
    lower_bound: float
    upper_bound: float
    mean_confidence: float = 0.0
    accuracy: float = 0.0
    count: int = 0


@dataclass
class CalibrationResult:
    """Result of calibration analysis."""
    prediction_type: str
    ece: float  # Expected Calibration Error
    mce: float  # Maximum Calibration Error
    brier_score: float
    bins: List[CalibrationBin]
    n_samples: int
    
    # Temperature scaling
    optimal_temperature: Optional[float] = None
    calibrated_ece: Optional[float] = None


class CalibrationValidator:
    """
    Validates calibration of ADAM's confidence scores.
    
    A well-calibrated system: when it says "80% confident",
    it should be correct 80% of the time.
    
    Key validations:
    1. Expected Calibration Error (ECE) < 0.05
    2. Maximum Calibration Error (MCE) < 0.15
    3. Reliability diagram analysis
    4. Temperature scaling effectiveness
    """
    
    def __init__(
        self,
        config: CalibrationValidationConfig,
        graph_client: 'Neo4jClient'
    ):
        self.config = config
        self.graph = graph_client
        self.n_bins = config.n_calibration_bins
        
        # Data storage
        self.predictions_by_type: Dict[str, List[Dict]] = {
            "personality": [],
            "mechanism": [],
            "state": [],
            "conversion": []
        }
    
    async def load_validation_data(
        self,
        start_date: datetime,
        end_date: datetime
    ) -> int:
        """Load prediction and outcome data for calibration validation."""
        # Load personality predictions with confidence
        personality_query = """
        MATCH (u:User)-[:HAS_PREDICTION]->(p:PersonalityPrediction)
        MATCH (u)-[:HAS_GROUND_TRUTH]->(gt:PersonalityGroundTruth)
        WHERE p.timestamp >= $start_date AND p.timestamp <= $end_date
        RETURN 
            p.confidence as confidence,
            p.openness as pred_o, p.conscientiousness as pred_c,
            p.extraversion as pred_e, p.agreeableness as pred_a, p.neuroticism as pred_n,
            gt.openness as true_o, gt.conscientiousness as true_c,
            gt.extraversion as true_e, gt.agreeableness as true_a, gt.neuroticism as true_n
        """
        
        results = await self.graph.execute_query(
            personality_query,
            {"start_date": start_date.isoformat(), "end_date": end_date.isoformat()}
        )
        
        for record in results:
            # Calculate accuracy as 1 - mean absolute error (normalized)
            errors = []
            for trait in ["o", "c", "e", "a", "n"]:
                pred = record.get(f"pred_{trait}")
                true = record.get(f"true_{trait}")
                if pred is not None and true is not None:
                    errors.append(abs(pred - true))
            
            if errors:
                mae = np.mean(errors)
                accuracy = 1 - mae / 4  # Normalize (max error is 4 on 1-5 scale)
                
                self.predictions_by_type["personality"].append({
                    "confidence": record["confidence"],
                    "accuracy": accuracy,
                    "correct": accuracy >= 0.75  # Threshold for "correct"
                })
        
        # Load conversion predictions
        conversion_query = """
        MATCH (decision:Decision)-[:RESULTED_IN]->(out:Outcome)
        WHERE decision.timestamp >= $start_date AND decision.timestamp <= $end_date
          AND decision.prediction_type = 'conversion'
        RETURN 
            decision.confidence as confidence,
            out.value as outcome_value
        """
        
        conv_results = await self.graph.execute_query(
            conversion_query,
            {"start_date": start_date.isoformat(), "end_date": end_date.isoformat()}
        )
        
        for record in conv_results:
            self.predictions_by_type["conversion"].append({
                "confidence": record["confidence"],
                "correct": record["outcome_value"] > 0
            })
        
        total = sum(len(preds) for preds in self.predictions_by_type.values())
        logger.info(f"Loaded {total} predictions for calibration validation")
        return total
    
    def calculate_calibration_metrics(
        self,
        predictions: List[Dict]
    ) -> CalibrationResult:
        """
        Calculate calibration metrics for a set of predictions.
        """
        if len(predictions) < 50:
            return CalibrationResult(
                prediction_type="unknown",
                ece=0.0,
                mce=0.0,
                brier_score=0.0,
                bins=[],
                n_samples=len(predictions)
            )
        
        confidences = np.array([p["confidence"] for p in predictions])
        correct = np.array([p["correct"] for p in predictions], dtype=float)
        
        # Create bins
        bin_boundaries = np.linspace(0, 1, self.n_bins + 1)
        bins = []
        
        for i in range(self.n_bins):
            lower = bin_boundaries[i]
            upper = bin_boundaries[i + 1]
            
            # Include upper bound in last bin
            if i == self.n_bins - 1:
                mask = (confidences >= lower) & (confidences <= upper)
            else:
                mask = (confidences >= lower) & (confidences < upper)
            
            bin_confidences = confidences[mask]
            bin_correct = correct[mask]
            
            if len(bin_confidences) > 0:
                bins.append(CalibrationBin(
                    lower_bound=lower,
                    upper_bound=upper,
                    mean_confidence=np.mean(bin_confidences),
                    accuracy=np.mean(bin_correct),
                    count=len(bin_confidences)
                ))
            else:
                bins.append(CalibrationBin(
                    lower_bound=lower,
                    upper_bound=upper,
                    count=0
                ))
        
        # Calculate ECE (Expected Calibration Error)
        total_samples = len(predictions)
        ece = sum(
            (b.count / total_samples) * abs(b.accuracy - b.mean_confidence)
            for b in bins if b.count > 0
        )
        
        # Calculate MCE (Maximum Calibration Error)
        calibration_errors = [
            abs(b.accuracy - b.mean_confidence) 
            for b in bins if b.count > 0
        ]
        mce = max(calibration_errors) if calibration_errors else 0
        
        # Calculate Brier Score
        brier = np.mean((confidences - correct) ** 2)
        
        return CalibrationResult(
            prediction_type="unknown",
            ece=ece,
            mce=mce,
            brier_score=brier,
            bins=bins,
            n_samples=len(predictions)
        )
    
    def apply_temperature_scaling(
        self,
        predictions: List[Dict],
        temperature: float
    ) -> List[Dict]:
        """
        Apply temperature scaling to calibrate confidence scores.
        
        scaled_confidence = sigmoid(logit(confidence) / temperature)
        """
        scaled = []
        
        for pred in predictions:
            conf = pred["confidence"]
            
            # Clip to avoid log(0)
            conf = np.clip(conf, 0.001, 0.999)
            
            # Apply temperature scaling
            logit = np.log(conf / (1 - conf))
            scaled_logit = logit / temperature
            scaled_conf = 1 / (1 + np.exp(-scaled_logit))
            
            scaled.append({
                "confidence": scaled_conf,
                "correct": pred["correct"]
            })
        
        return scaled
    
    def find_optimal_temperature(
        self,
        predictions: List[Dict]
    ) -> Tuple[float, float]:
        """
        Find optimal temperature for calibration via grid search.
        
        Returns (optimal_temperature, calibrated_ece)
        """
        best_temp = 1.0
        best_ece = float('inf')
        
        for temp in np.arange(0.5, 3.0, 0.1):
            scaled = self.apply_temperature_scaling(predictions, temp)
            result = self.calculate_calibration_metrics(scaled)
            
            if result.ece < best_ece:
                best_ece = result.ece
                best_temp = temp
        
        return best_temp, best_ece

    ) -> ValidationResult:
        """

"""
Neo4j query utilities for validation data access.
"""

from datetime import datetime
from typing import Dict, List, Optional, Any


class ValidationGraphQueries:
    """
    Query utilities for validation data in Neo4j.
    """
    
    @staticmethod
    async def store_validation_suite_result(
        graph_client: 'Neo4jClient',
        result: ValidationSuiteResult
    ) -> str:
        """Store complete validation suite result in graph."""
        
        # Create suite run node
        query = """
        CREATE (run:ValidationSuiteRun {
            suite_id: $suite_id,
            suite_name: $suite_name,
            run_timestamp: datetime($run_timestamp),
            overall_status: $overall_status,
            critical_failures: $critical_failures,
            warnings: $warnings,
            duration_seconds: $duration_seconds
        })
        RETURN run.suite_id
        """
        
        await graph_client.execute_query(query, {
            "suite_id": result.suite_id,
            "suite_name": result.suite_name,
            "run_timestamp": result.run_timestamp.isoformat(),
            "overall_status": result.overall_status.value,
            "critical_failures": result.critical_failures,
            "warnings": result.warnings,
            "duration_seconds": result.total_duration_seconds
        })
        
        # Store mechanism results
        if result.mechanism_results:
            for mechanism, mech_result in result.mechanism_results.items():
                mech_query = """
                MATCH (run:ValidationSuiteRun {suite_id: $suite_id})
                CREATE (r:MechanismValidationResult {
                    result_id: $result_id,
                    mechanism: $mechanism,
                    overall_effectiveness: $effectiveness,
                    relative_lift: $lift,
                    lift_p_value: $p_value,
                    matched_conversion_rate: $matched_rate,
                    control_conversion_rate: $control_rate,
                    validation_status: $status,
                    validated_at: datetime()
                })
                CREATE (run)-[:INCLUDES_MECHANISM_RESULT]->(r)
                """
                
                await graph_client.execute_query(mech_query, {
                    "suite_id": result.suite_id,
                    "result_id": f"{result.suite_id}_{mechanism}",
                    "mechanism": mechanism,
                    "effectiveness": mech_result.overall_effectiveness,
                    "lift": mech_result.relative_lift,
                    "p_value": mech_result.lift_p_value,
                    "matched_rate": mech_result.matched_conversion_rate,
                    "control_rate": mech_result.control_conversion_rate,
                    "status": mech_result.validation_status.value
                })
        
        # Link to previous run
        link_query = """
        MATCH (current:ValidationSuiteRun {suite_id: $suite_id})
        MATCH (previous:ValidationSuiteRun)
        WHERE previous.suite_name = $suite_name
          AND previous.suite_id <> $suite_id
        WITH current, previous
        ORDER BY previous.run_timestamp DESC
        LIMIT 1
        CREATE (current)-[:PREVIOUS_RUN]->(previous)
        """
        
        await graph_client.execute_query(link_query, {
            "suite_id": result.suite_id,
            "suite_name": result.suite_name
        })
        
        return result.suite_id
    
    @staticmethod
    async def get_validation_trend(
        graph_client: 'Neo4jClient',
        suite_name: str,
        days: int = 30
    ) -> List[Dict]:
        """Get validation trend over time."""
        query = """
        MATCH (run:ValidationSuiteRun)
        WHERE run.suite_name = $suite_name
          AND run.run_timestamp >= datetime() - duration('P' + $days + 'D')
        RETURN run.run_timestamp AS timestamp,
               run.overall_status AS status,
               run.critical_failures AS failures,
               run.duration_seconds AS duration
        ORDER BY run.run_timestamp DESC
        """
        
        results = await graph_client.execute_query(query, {
            "suite_name": suite_name,
            "days": str(days)
        })
        
        return [dict(r) for r in results]
    
    @staticmethod
    async def get_mechanism_effectiveness_trend(
        graph_client: 'Neo4jClient',
        mechanism: str,
        days: int = 90
    ) -> List[Dict]:
        """Get mechanism effectiveness over time."""
        query = """
        MATCH (r:MechanismValidationResult)
        WHERE r.mechanism = $mechanism
          AND r.validated_at >= datetime() - duration('P' + $days + 'D')
        RETURN r.validated_at AS timestamp,
               r.relative_lift AS lift,
               r.lift_p_value AS p_value,
               r.validation_status AS status
        ORDER BY r.validated_at DESC
        """
        
        results = await graph_client.execute_query(query, {
            "mechanism": mechanism,
            "days": str(days)
        })
        
        return [dict(r) for r in results]
    
    @staticmethod
    async def get_ground_truth_statistics(
        graph_client: 'Neo4jClient'
    ) -> Dict[str, Any]:
        """Get statistics about available ground truth data."""
        query = """
        MATCH (gt:PersonalityGroundTruth)
        WHERE gt.is_valid = true
        WITH count(gt) AS total_valid,
             avg(gt.internal_consistency_alpha) AS mean_alpha,
             collect(DISTINCT gt.instrument) AS instruments
        MATCH (mgt:MechanismEffectivenessGroundTruth)
        WITH total_valid, mean_alpha, instruments,
             count(mgt) AS mechanism_records,
             count(DISTINCT mgt.mechanism) AS mechanisms_covered
        RETURN {
            personality_ground_truth: total_valid,
            mean_internal_consistency: mean_alpha,
            instruments_used: instruments,
            mechanism_effectiveness_records: mechanism_records,
            mechanisms_with_data: mechanisms_covered
        } AS stats
        """
        
        results = await graph_client.execute_query(query, {})
        return results[0]["stats"] if results else {}

@router.post("/suites/run", response_model=ValidationSuiteResponse)
async def run_validation_suite(
    request: ValidationSuiteRequest,
    background_tasks: BackgroundTasks,
    orchestrator: ValidationOrchestrator = Depends(get_orchestrator)
):
    """
    Trigger a validation suite run.
    
    By default runs asynchronously and returns immediately with a suite_id
    that can be polled for status.
    """
    suite_id = str(uuid.uuid4())
    
    # Set default date range
    end_date = request.end_date or datetime.utcnow()
    start_date = request.start_date or (end_date - timedelta(days=30))
    
    if request.async_execution:
        # Run in background
        background_tasks.add_task(
            orchestrator.run_validation_suite,
            start_date,
            end_date
        )
        
        return ValidationSuiteResponse(
            suite_id=suite_id,
            suite_name=request.suite_name,
            status="started",
            message="Validation suite started. Poll /suites/{suite_id}/status for progress.",
            estimated_duration_seconds=300
        )
    else:
        # Run synchronously
        result = await orchestrator.run_validation_suite(start_date, end_date)
        
        return ValidationSuiteResponse(
            suite_id=result.suite_id,
            suite_name=result.suite_name,
            status=result.overall_status.value,
            message=f"Completed in {result.total_duration_seconds:.1f}s"
        )

         CASE WHEN gt.user_personality.openness >= 3.5 THEN 'high'
              WHEN gt.user_personality.openness <= 2.5 THEN 'low'
              ELSE 'medium' END AS o_level,
         CASE WHEN gt.user_personality.neuroticism >= 3.5 THEN 'high'
              WHEN gt.user_personality.neuroticism <= 2.5 THEN 'low'
              ELSE 'medium' END AS n_level
    RETURN o_level, n_level,
           avg(CASE WHEN gt.outcome_value > 0 THEN 1.0 ELSE 0.0 END) AS conversion_rate,
           count(*) AS sample_size
    ORDER BY conversion_rate DESC
    """
    
    results = await graph_client.execute_query(query, {"mechanism": mechanism})
    
    return {
        "mechanism": mechanism,
        "segments": [dict(r) for r in results]
    }

    graph_client: Neo4jClient = Depends(get_graph_client)
):
    """Get ranking of intelligence sources by effectiveness."""
    query = """
    MATCH (r:FusionValidationResult)
    RETURN r.source_accuracies AS accuracies,
           r.source_contributions AS contributions,
           r.ablation_results AS ablation
    ORDER BY r.validated_at DESC
    LIMIT 1
    """
    
    results = await graph_client.execute_query(query, {})
    
    if not results:
        return {"sources": []}
    
    data = results[0]
    accuracies = data["accuracies"] or {}
    contributions = data["contributions"] or {}
    ablation = data["ablation"] or {}
    
    # Combine into ranked list
    sources = []
    for source in accuracies.keys():
        sources.append({
            "source": source,
            "accuracy": accuracies.get(source, 0),
            "contribution": contributions.get(source, 0),
            "value_when_removed": ablation.get(source, 0)
        })
    
    # Sort by accuracy
    sources.sort(key=lambda x: x["accuracy"], reverse=True)
    
    return {"sources": sources}

    """Get comprehensive validation dashboard data."""
    # Get latest run
    latest_query = """
    MATCH (run:ValidationSuiteRun)
    RETURN run
    ORDER BY run.run_timestamp DESC
    LIMIT 1
    """
    
    latest_results = await graph_client.execute_query(latest_query, {})
    
    if not latest_results:
        return ValidationDashboardResponse(
            last_run_timestamp=None,
            last_run_status="never_run",
            mechanisms_validated=0,
            mechanisms_passed=0,
            state_trait_valid=False,
            fusion_valid=False,
            fairness_issues=[],
            calibration_status={},
            days_since_failure=0
        )
    
    run = latest_results[0]["run"]
    
    # Get mechanism counts
    mech_query = """
    MATCH (run:ValidationSuiteRun {suite_id: $suite_id})-[:INCLUDES_MECHANISM_RESULT]->(r)
    RETURN count(r) AS total,
           sum(CASE WHEN r.validation_status = 'passed' THEN 1 ELSE 0 END) AS passed
    """
    
    mech_results = await graph_client.execute_query(mech_query, {"suite_id": run["suite_id"]})
    mech_data = mech_results[0] if mech_results else {"total": 0, "passed": 0}
    
    # Get fairness issues
    fairness_query = """
    MATCH (r:FairnessValidationResult)
    WHERE r.validation_status <> 'passed'
    RETURN r.flagged_issues AS issues
    ORDER BY r.validated_at DESC
    LIMIT 5
    """
    
    fairness_results = await graph_client.execute_query(fairness_query, {})
    fairness_issues = []
    for r in fairness_results:
        fairness_issues.extend(r.get("issues", [])[:3])
    
    # Days since failure
    failure_query = """
    MATCH (run:ValidationSuiteRun)
    WHERE run.overall_status = 'failed'
    RETURN run.run_timestamp AS failure_time
    ORDER BY run.run_timestamp DESC
    LIMIT 1
    """
    
    failure_results = await graph_client.execute_query(failure_query, {})
    
    if failure_results:
        failure_time = failure_results[0]["failure_time"]
        days_since = (datetime.utcnow() - failure_time).days
    else:
        days_since = 999  # Never failed
    
    return ValidationDashboardResponse(
        last_run_timestamp=run["run_timestamp"],
        last_run_status=run.get("overall_status", "unknown"),
        mechanisms_validated=mech_data["total"],
        mechanisms_passed=mech_data["passed"],
        state_trait_valid=True,  # Would query actual result
        fusion_valid=True,  # Would query actual result
        fairness_issues=fairness_issues[:5],
        calibration_status={},  # Would query actual results
        days_since_failure=days_since
    )

class ValidationMetricsCollector:
    """
    Collects and publishes validation metrics to Prometheus.
    """
    
    def __init__(self, metrics_client: 'MetricsClient'):
        self.client = metrics_client
    
    def record_suite_result(self, result: ValidationSuiteResult):
        """Record metrics from a validation suite run."""
        suite_name = result.suite_name
        
        # Suite-level metrics
        status_value = {
            ValidationStatus.PASSED: 1.0,
            ValidationStatus.WARNING: 0.5,
            ValidationStatus.FAILED: 0.0
        }.get(result.overall_status, -1.0)
        
        validation_suite_status.labels(suite_name=suite_name).set(status_value)
        validation_suite_runs_total.labels(
            suite_name=suite_name, 
            status=result.overall_status.value
        ).inc()
        validation_suite_duration_seconds.labels(suite_name=suite_name).observe(
            result.total_duration_seconds
        )
        validation_critical_failures.labels(suite_name=suite_name).set(
            len(result.critical_failures)
        )
        validation_warnings.labels(suite_name=suite_name).set(
            len(result.warnings)
        )
        
        # Mechanism metrics
        if result.mechanism_results:
            for mechanism, mech_result in result.mechanism_results.items():
                mech_status = {
                    ValidationStatus.PASSED: 1.0,
                    ValidationStatus.WARNING: 0.5,
                    ValidationStatus.FAILED: 0.0
                }.get(mech_result.validation_status, -1.0)
                
                mechanism_validation_status.labels(mechanism=mechanism).set(mech_status)
                mechanism_lift.labels(mechanism=mechanism).set(mech_result.relative_lift)
                mechanism_lift_p_value.labels(mechanism=mechanism).set(mech_result.lift_p_value)
                mechanism_matched_conversion_rate.labels(mechanism=mechanism).set(
                    mech_result.matched_conversion_rate
                )
                mechanism_control_conversion_rate.labels(mechanism=mechanism).set(
                    mech_result.control_conversion_rate
                )
                mechanism_sample_size.labels(mechanism=mechanism, group="matched").set(
                    mech_result.matched_exposures
                )
                mechanism_sample_size.labels(mechanism=mechanism, group="control").set(
                    mech_result.control_exposures
                )
        
        # State Ã— Trait metrics
        if result.state_trait_result:
            st = result.state_trait_result
            state_trait_r_squared.labels(model_type="trait_only").set(st.trait_only_r_squared)
            state_trait_r_squared.labels(model_type="trait_state").set(st.trait_state_r_squared)
            state_trait_incremental_r_squared.set(st.incremental_r_squared)
            state_trait_interaction_count.set(len(st.significant_interactions))
            
            for state, contrib in st.state_contributions.items():
                if isinstance(contrib, dict) and "correlation" in contrib:
                    state_contribution.labels(state=state).set(contrib["correlation"])
        
        # Fusion metrics
        if result.fusion_result:
            f = result.fusion_result
            fusion_accuracy.set(f.fusion_accuracy)
            fusion_improvement.set(f.fusion_improvement)
            source_agreement_rate.set(f.mean_agreement_rate)
            
            for source, acc in f.source_accuracies.items():
                single_source_accuracy.labels(source=source).set(acc)
            
            for source, weight in f.source_contributions.items():
                source_contribution_weight.labels(source=source).set(weight)
        
        # Temporal metrics
        if result.temporal_results:
            for construct, temp_result in result.temporal_results.items():
                if temp_result.construct_type == "trait" and temp_result.trait_icc:
                    trait_icc.labels(trait=construct).set(temp_result.trait_icc)
                    
                    for interval, corr in temp_result.retest_correlations.items():
                        trait_test_retest.labels(
                            trait=construct,

                elif temp_result.construct_type == "state" and temp_result.state_variability:

                drift = 1.0 if temp_result.drift_detected else 0.0

        # Fairness metrics

            for attr, fair_result in result.fairness_results.items():
                attr_name = attr.value if hasattr(attr, 'value') else str(attr)
                
                if fair_result.disparate_impact_ratio:
                    disparate_impact_ratio.labels(attribute=attr_name).set(

                        temp = cal_result.secondary_metrics["optimal_temperature"]
                        if temp:
                            calibration_temperature.labels(prediction_type=pred_type).set(temp)

        predictions = []
        for _ in range(1000):

            predictions.append({"confidence": conf, "correct": correct})

        for i in range(1000):

                "confidence": 0.9,  # Always confident
                "correct": i % 2 == 0  # Only 50% correct
            })
        
        result = validator.calculate_calibration_metrics(predictions)
        
        # Should have high ECE (overconfident)
        assert result.ece > 0.3
